Portal Plugins

Start by creating a Layout with the base code: 

UI.plugin (function() {  
    console.log ('Custom plugin 2.1.8');  
});

And then deploy it via npm run build
In development mode use npm run watch have changes automatically deployed on file save

Customizing the header

var toolbar = UI.header.addToolbar('Help');  
toolbar.add('Help', 'help_action')  
toolbar.add('Intranet', 'intranet_action')

Or to add a tool under the tool menu: 

UI.header.addTool('Help', 'help_action');

Customizing the sidebar

var menu = UI.sidebar.addMenu('Menu 1', 'icon');  
menu.add('Shortcut', function() {console.log('onclick')});
menu.add('Shortcut 1','#node/Node Path/division');  
menu.add('Shortcut 2','#node/Node Path/division', 'folder');  

UI.sidebar.addText('<b>Hellow World</b><p>Just some brief text in the sidebar to get you started<p/>');

Customizing the Document View Tabs

Tabs are customized using Handlerbars (.hbs) templates
to change the default from DocumentView to something else: 

router.doc.template = 'Simple';

To change the template for documents in a specific node: 

UI.addTemplate( {  
    node: 'Division/cabinet/node', // the node for which this template should be used  
    template: 'NewTemplateName' // the name of the Handlerbars template  
});

To add a new tab: 

UI.addTab({  
    title: 'Tab Name', // can include HTML for icons  
    name: 'newtab', // the name under which the model data will be placed for use in the template  
    template: 'NewTab.hbs', // the Handlebars template to use to render the contents of the tab  
    filter: function(doc) {return true}, //  gets called to check wether to show the tab or not  
    load: function() {return this.getHistory() } // a function that gets called when the tab is clicked -  
     //used to load additional data and/or inject new values for the template to render  
})

Adding an Upload Button

You can add a button that will upload a document to a specific node with preset indexes, using the: following: 

<div class="col-md-4">  
  <button class="btn btn-default light"   
    data-upload="/temp/file"   
    data-action="upload_direct"  
    data-node="Supprting Documents/"  <!-- Required -->  
    data-index.formNo="{{formNo}}"   
    data-index.doc_type="Doc Type 1"  
    data-toggle="button" >  
       <span class="icon-upload"> </span> Upload Supporting Document  
   </button>  
</div>

data-node is a required path to the node to place the upload document
data-index.[index name]is one or more custom indexes to set on the node

The button must be placed inside a div to properly register an upload handler

There are 3 types of data-action that can be used for upload handlers:

upload_direct will prompt the user to select a file and then import the document immediately
upload will prompt for a file and then allow the user to index the document according to the destination node
create will create an Item instead of prompting for a file  create_direct will prompt for indexes and then create an Item

Uploading and appending to a list

First create a new tab to contain the list:

UI.addTab({  
    title: 'Supporting Docs',
    template: 'Supporting.hbs',
    load: function() {return PQL.query("SELECT * FROM 'Node/Supporting' WHERE formNo = '" + this.docId + "'") }
})

And the template:

<div class="row" >
    <topbar-left></topbar-left>
    <topbar-right>
                <div class="btn-group">
                    <button class="btn btn-default  btn-sm light upload" data-upload="/temp/file"
                            data-action="upload_direct"
                            data-append="supporting"
                            data-node="Node/Supporting"
                            data-toggle="button">
                        <span class="icon-upload"> </span> Upload
                    </button>
                </div>
    </topbar-right>
</div>

<content-center>
    {{> grid supporting}}
</content-center>

data-append will append the uploaded document or item to the specified document list
{{> grid supporting}} will render the supporting document list

Enhancing the Document Model

If you need to add fields and methods that need to be available in templates you can add a document filter - The filter will be called just before being returned and can update, add or delete indexes and fields. 

PaperTrail.addDocumentFilter(doc => {  
    doc.indexes.total = parseInt(doc.get('unit')) * parseInt(doc.get('price'));  
    return doc;  
});

addDocumentFilter() can also return a promise so you can pull in data if necessary eg. 

PaperTrail.addDocumentFilter(doc => {  
        // populate an approvers array based on approver_0-4 indexes and corresponding audits by those users  
        return doc.getHistory().then((history) => {  
            doc.approvers = [];  
            _.each(['0', '1', '2', '3'], (it) => {  
                if (is.empty(doc.get('approver_' + it))) {  
                    return;  
                }  
                var approver = {  
                    approver: doc.get('approver_' + it),  
                    status: 'Pending',  
                    comment: ''  
                };  
                var audit = _.find(doc.history, (it) => {  
                    var type = it.auditType.toLowerCase();  
                    return it.createdBy == approver.approver && (type.contains('sign') || type.contains('approve') || type.contains('reject') || type.contains('rework'));  
                });  
                if (audit != null) {  
                    approver.status = audit.auditType;  
                    approver.comment = audit.audit;  
                }  
                doc.approvers.push(approver)  
            });  
        })  

    });  
})
